```typescript
import type { NextRequest } from 'next/server'
import { detectInputType, isPrivateIp } from '@/lib/detect-input-type'
```

Export an `async function GET(request: NextRequest)` that:
1. Reads `q` from `request.nextUrl.searchParams.get('q')`.
2. If `!q`, return 422 with `INVALID_INPUT`.
3. If `isPrivateIp(q)`, return 422 with `PRIVATE_IP`.
4. Define `const RECORD_TYPES = ['A', 'AAAA', 'MX', 'TXT', 'NS', 'CNAME'] as const`.
5. Use `Promise.allSettled` to fetch all 6 record types in parallel:
   ```typescript
   const results = await Promise.allSettled(
     RECORD_TYPES.map(async (type) => {
       const res = await fetch(
         `https://dns.google/resolve?name=${encodeURIComponent(q)}&type=${type}`,
         { cache: 'no-store' }
       )
       if (!res.ok) throw new Error(`DNS ${type} fetch failed: ${res.status}`)
       const data = await res.json()
       return { type, answers: data.Answer ?? [] }
     })
   )
   ```
6. Transform results into an object keyed by record type:
   ```typescript
   const records: Record<string, Array<{ name: string; data: string; TTL: number }>> = {}
   for (const [i, result] of results.entries()) {
     const type = RECORD_TYPES[i]
     if (result.status === 'fulfilled') {
       records[type] = result.value.answers.map((a: any) => ({
         name: a.name,
         data: a.data,
         TTL: a.TTL,
       }))
     } else {
       records[type] = []
     }
   }
   ```
7. Return `Response.json({ records })`.
8. Wrap outer logic in try/catch. On catch, return 502 with `UPSTREAM_ERROR`.

```typescript
import type { NextRequest } from 'next/server'
import { detectInputType, isPrivateIp, buildIPv4PtrName, buildIPv6PtrName } from '@/lib/detect-input-type'
```

Export an `async function GET(request: NextRequest)` that:
1. Reads `q` from `request.nextUrl.searchParams.get('q')`.
2. If `!q`, return 422 with `INVALID_INPUT`.
3. If `isPrivateIp(q)`, return 422 with `PRIVATE_IP`.
4. Determine input type via `detectInputType(q)`.
5. If type is `'domain'`, return 422 with `{ error: { code: 'INVALID_INPUT', message: 'Reverse DNS requires an IP address, not a domain.' } }`.
6. Build PTR name:
   - If `'ipv4'`: `buildIPv4PtrName(q)` (returns e.g. `8.8.8.8.in-addr.arpa`)
   - If `'ipv6'`: `buildIPv6PtrName(q)` (returns e.g. `8.8.8.8.0...2.ip6.arpa`)
7. Fetch `https://dns.google/resolve?name=${encodeURIComponent(ptrName)}&type=PTR` with `{ cache: 'no-store' }`.
8. If `!res.ok`, return 502 with `UPSTREAM_ERROR`.
9. Parse JSON. Guard with `data.Answer ?? []`.
10. Extract PTR hostnames: `const ptrs = (data.Answer ?? []).map((a: any) => a.data)`.
11. Return `Response.json({ ptr: ptrs })`.
12. Wrap in try/catch. On catch, return 502.

**Create `src/app/api/whois/route.ts`:**

```typescript
import type { NextRequest } from 'next/server'
import { detectInputType, isPrivateIp } from '@/lib/detect-input-type'
```

Export an `async function GET(request: NextRequest)` that:
1. Reads `q` from `request.nextUrl.searchParams.get('q')`.
2. If `!q`, return 422 with `INVALID_INPUT`.
3. If `isPrivateIp(q)`, return 422 with `PRIVATE_IP`.
4. Determine input type via `detectInputType(q)`.
5. Build RDAP URL:
   - If `'domain'`: `https://rdap.org/domain/${encodeURIComponent(q)}`
   - If `'ipv4'` or `'ipv6'`: `https://rdap.org/ip/${encodeURIComponent(q)}`
   - If `'unknown'`: return 422 with `INVALID_INPUT`.
6. Fetch the RDAP URL with `{ cache: 'no-store' }`. Node.js fetch follows 302 redirects automatically (rdap.org is a bootstrap that 302s to the authoritative RDAP server).
7. If `res.status === 404`, return `Response.json({ error: { code: 'RDAP_NOT_SUPPORTED', message: 'WHOIS not available via RDAP for this TLD.' } }, { status: 404 })`.
8. If `!res.ok`, return 502 with `UPSTREAM_ERROR` and `upstreamStatus: res.status`.
9. Parse the RDAP JSON. Extract and normalize into a flat response object:
   ```typescript
   const rdap = await res.json()
   const response = {
     name: rdap.name ?? rdap.handle ?? q,
     registrar: rdap.entities?.find((e: any) => e.roles?.includes('registrar'))?.vcardArray?.[1]?.find((v: any) => v[0] === 'fn')?.[3] ?? null,
     creationDate: rdap.events?.find((e: any) => e.eventAction === 'registration')?.eventDate ?? null,
     expirationDate: rdap.events?.find((e: any) => e.eventAction === 'expiration')?.eventDate ?? null,
     lastChanged: rdap.events?.find((e: any) => e.eventAction === 'last changed')?.eventDate ?? null,
     status: rdap.status ?? [],
     nameservers: rdap.nameservers?.map((ns: any) => ns.ldhName) ?? [],
     raw: rdap,
   }
   ```
10. Return `Response.json(response)`.
11. Wrap in try/catch. On catch, return 502 with message `'Failed to reach WHOIS API'`.
