File contents:

```tsx
'use client'

import { useState } from 'react'
import { Copy, Check } from 'lucide-react'
import { Button } from '@/components/ui/button'

interface CopyButtonProps {
  value: string
  className?: string
}

export function CopyButton({ value, className }: CopyButtonProps) {
  const [copied, setCopied] = useState(false)

  const handleCopy = () => {
    navigator.clipboard.writeText(value).then(() => {
      setCopied(true)
      setTimeout(() => setCopied(false), 1500)
    })
  }

  return (
    <Button
      variant="ghost"
      size="icon"
      className={`size-6 opacity-0 group-hover:opacity-100 transition-opacity duration-150 ${className ?? ''}`}
      onClick={handleCopy}
      aria-label="Copy value"
    >
      {copied
        ? <Check className="size-3" />
        : <Copy className="size-3" />}
    </Button>
  )
}
```

Key decisions per CONTEXT.md:
- `navigator.clipboard.writeText` — no third-party library (D-11)
- Icon swaps to Check for 1500ms then reverts (D-10)
- `opacity-0 group-hover:opacity-100` — appears on hover only (D-09)
- `transition-opacity duration-150` — smooth transition (Claude's discretion)
- `size-6` button, `size-3` icon (UI-SPEC Spacing Scale)
- `variant="ghost"` — no background, no border (consistent with UI-SPEC)
- `aria-label="Copy value"` (UI-SPEC Copywriting Contract)
